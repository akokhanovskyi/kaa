85 template <ChannelType Type>
86 void AbstractHttpChannel<Type>::processTypes(const std::map<TransportType, ChannelDirection>& types, boost::unique_lock<boost::mutex>& lock)
87 {
88  AbstractServerInfoPtr server = currentServer_;
89 
90  const auto& bodyRaw = multiplexer_->compileRequest(types);
91  // Creating HTTP request using the given data
92  boost::shared_ptr<IHttpRequest> postRequest = createRequest(server, bodyRaw);
93 
94  KAA_MUTEX_UNLOCKING("channelGuard_");
95  lock.unlock();
96  KAA_MUTEX_UNLOCKED("channelGuard_");
97  try {
98  // Sending http request
99  auto response = httpClient_.sendRequest(*postRequest);
100 
101  KAA_MUTEX_LOCKING("channelGuard_");
102  boost::unique_lock<boost::mutex> lockInternal(channelGuard_);
103  KAA_MUTEX_LOCKED("channelGuard_");
104 
105  // Retrieving the avro data from the HTTP response
106  const std::string& processedResponse = retrieveResponse(*response);
107  lastConnectionFailed_ = false;
108 
109  KAA_MUTEX_UNLOCKING("channelGuard_");
110  lockInternal.unlock();
111  KAA_MUTEX_UNLOCKED("channelGuard_");
112 
113  if (!processedResponse.empty()) {
114  demultiplexer_->processResponse(
115  std::vector<boost::uint8_t>(reinterpret_cast<const boost::uint8_t *>(processedResponse.data()),
116  reinterpret_cast<const boost::uint8_t *>(processedResponse.data() + processedResponse.size())));
117  }
118  } catch (std::exception& e) {
119  KAA_LOG_ERROR(boost::format("Connection failed, server %1%:%2%: %3%") % server->getHost() % server->getPort() % e.what());
120 
121  KAA_MUTEX_LOCKING("channelGuard_");
122  boost::unique_lock<boost::mutex> lockInternal(channelGuard_);
123  KAA_MUTEX_LOCKED("channelGuard_");
124 
125  lastConnectionFailed_ = true;
126 
127  KAA_MUTEX_UNLOCKING("channelGuard_");
128  lockInternal.unlock();
129  KAA_MUTEX_UNLOCKED("channelGuard_");
130 
131  channelManager_->onServerFailed(server);
132  }
133 }

135 template <ChannelType Type>
136 void AbstractHttpChannel<Type>::sync(TransportType type)
137 {
138  const auto& supportedTypes = getSupportedTransportTypes();
139  auto it = supportedTypes.find(type);
140  if (it != supportedTypes.end() && (it->second == ChannelDirection::UP || it->second == ChannelDirection::BIDIRECTIONAL)) {
141  KAA_MUTEX_LOCKING("channelGuard_");
142  boost::unique_lock<boost::mutex> lock(channelGuard_);
143  KAA_MUTEX_LOCKED("channelGuard_");
144  if (currentServer_) {
145  processTypes(std::map<TransportType, ChannelDirection>({ { type, it->second } }), lock);
146  } else {
147  lastConnectionFailed_ = true;
148  KAA_LOG_WARN(boost::format("Can't sync channel %1%. Server is null") % getId());
149  }
150  } else {
151  KAA_LOG_ERROR(boost::format("Unsupported transport type for channel %1%") % getId());
152  }
153 }
