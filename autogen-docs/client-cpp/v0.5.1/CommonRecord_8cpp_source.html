88 avro::GenericDatum CommonRecord::toAvro() const
89 {
90  avro::GenericDatum datum(getSchema());
91  avro::GenericRecord &record = datum.value<avro::GenericRecord>();
92 
93  size_t fieldsCount = schema_->names();
94  for (size_t fieldIndex = 0; fieldIndex < fieldsCount; ++fieldIndex) {
95  const std::string & fieldName = schema_->nameAt(fieldIndex);
96  auto it = getFields().find(fieldName);
97  if (it != getFields().end()) {
98  const avro::NodePtr &innode = schema_->leafAt(fieldIndex);
99 
100  avro::GenericDatum fieldDatum = it->second->toAvro();
101  if (innode->type() == avro::AVRO_UNION) {
102 
103  avro::Type avro_type = fieldDatum.type();
104 
105  avro::GenericDatum indatum(innode);
106  size_t branches = indatum.unionBranch();
107  size_t currentFieldBranch = 0;
108 
109  bool is_compound = avro::isCompound(avro_type);
110 
111  while (currentFieldBranch < branches) {
112  avro::Type intype = innode->leafAt(currentFieldBranch)->type();
113  indatum.selectBranch(currentFieldBranch);
114 
115  bool found = false;
116  if (is_compound) {
117  ISchemaDependent * dependent = dynamic_cast<ISchemaDependent *>(it->second.get());
118  if (innode->leafAt(currentFieldBranch)->hasName() && dependent->getSchema()->hasName() &&
119  dependent->getSchema()->name().fullname() == innode->leafAt(currentFieldBranch)->name().fullname()) {
120  found = SetAvroValueHelper::setValue(avro_type, indatum, fieldDatum);
121  } else if (innode->leafAt(currentFieldBranch)->type() == avro::Type::AVRO_ARRAY
122  && dependent->getSchema()->type() == avro::Type::AVRO_ARRAY) {
123  found = SetAvroValueHelper::setValue(avro::Type::AVRO_ARRAY, indatum, fieldDatum);
124  }
125  } else if (intype == avro_type) {
126  found = SetAvroValueHelper::setValue(avro_type, indatum, fieldDatum);
127  }
128  if (found) {
129  break;
130  }
131  ++currentFieldBranch;
132  }
133  record.setField(it->first, indatum);
134  } else {
135  record.setField(it->first, fieldDatum);
136  }
137  }
138  }
139  return datum;
140 }
