85 void DefaultOperationLongPollChannel::executeTask()
86 {
87  KAA_MUTEX_LOCKING("channelGuard_");
88  boost::unique_lock<boost::mutex> lock(channelGuard_);
89  KAA_MUTEX_LOCKED("channelGuard_");
90  taskPosted_ = false;
91  if (stopped_) {
92  return;
93  }
94  connectionInProgress_ = true;
95 
96  OperationServerLongPollInfoPtr server = currentServer_;
97 
98  const auto& bodyRaw = multiplexer_->compileRequest(getSupportedTransportTypes());
99  // Creating HTTP request using the given data
100  boost::shared_ptr<IHttpRequest> postRequest = httpDataProcessor_.createOperationRequest(server->getUrl(), bodyRaw);
101 
102  KAA_MUTEX_UNLOCKING("channelGuard_");
103  lock.unlock();
104  KAA_MUTEX_UNLOCKED("channelGuard_");
105  try {
106  // Sending http request
107  auto response = httpClient_.sendRequest(*postRequest);
108  KAA_MUTEX_LOCKING("channelGuard_");
109  boost::unique_lock<boost::mutex> lockInternal(channelGuard_);
110  KAA_MUTEX_LOCKED("channelGuard_");
111  // Retrieving the avro data from the HTTP response
112  connectionInProgress_ = false;
113  const std::string& processedResponse = httpDataProcessor_.retrieveOperationResponse(*response);
114  KAA_MUTEX_UNLOCKING("channelGuard_");
115  lockInternal.unlock();
116  KAA_MUTEX_UNLOCKED("channelGuard_");
117  demultiplexer_->processResponse(
118  std::vector<boost::uint8_t>(reinterpret_cast<const boost::uint8_t *>(processedResponse.data()),
119  reinterpret_cast<const boost::uint8_t *>(processedResponse.data() + processedResponse.size())));
120  waitCondition_.notify_all();
121  } catch (std::exception& e) {
122  KAA_MUTEX_LOCKING("channelGuard_");
123  boost::unique_lock<boost::mutex> lockException(channelGuard_);
124  KAA_MUTEX_LOCKED("channelGuard_");
125 
126  bool isServerFailed = false;
127  connectionInProgress_ = false;
128  if (stopped_) {
129  KAA_LOG_INFO(boost::format("Connection for channel %1% was aborted") % getId());
130  } else {
131  KAA_LOG_ERROR(boost::format("Connection failed, server %1%:%2%: %3%") % server->getHost() % server->getPort() % e.what());
132  isServerFailed = true;
133  stopped_ = true;
134  }
135  KAA_MUTEX_UNLOCKING("channelGuard_");
136  lockException.unlock();
137  KAA_MUTEX_UNLOCKED("channelGuard_");
138 
139  waitCondition_.notify_all();
140  if (isServerFailed) {
141  channelManager_->onServerFailed(server);
142  }
143  return;
144  }
145 
146  KAA_MUTEX_LOCKING("channelGuard_");
147  lock.lock();
148  KAA_MUTEX_LOCKED("channelGuard_");
149  if (!stopped_ && !taskPosted_) {
150  postTask();
151  }
152 }

154 void DefaultOperationLongPollChannel::sync(TransportType type)
155 {
156  auto it = SUPPORTED_TYPES.find(type);
157  if (it != SUPPORTED_TYPES.end() && (it->second == ChannelDirection::UP || it->second == ChannelDirection::BIDIRECTIONAL)) {
158  KAA_MUTEX_LOCKING("channelGuard_");
159  boost::unique_lock<boost::mutex> lock(channelGuard_);
160  KAA_MUTEX_LOCKED("channelGuard_");
161  if (currentServer_) {
162  KAA_MUTEX_UNLOCKING("channelGuard_");
163  lock.unlock();
164  KAA_MUTEX_UNLOCKED("channelGuard_");
165 
166  stopPoll();
167  startPoll();
168  } else {
169  KAA_LOG_WARN(boost::format("Can't sync channel %1%. Server is null") % getId());
170  }
171  } else {
172  KAA_LOG_ERROR(boost::format("Unsupported transport type for channel %1%") % getId());
173  }
