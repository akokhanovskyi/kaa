85 void BootstrapManager::useNextOperationsServer(ChannelType type)
86 {
87  boost::recursive_mutex::scoped_lock lock(guard_);
88  auto it = operationServersIterators_.find(type);
89  auto collectionIt = operationServers_.find(type);
90  if (it != operationServersIterators_.end() && collectionIt != operationServers_.end()) {
91  if (++(it->second) != collectionIt->second.size()) {
92  KAA_LOG_INFO(boost::format("Going to switch to server: %1% for channel type: %2%")
93  % LoggingUtils::OperationServerToString(collectionIt->second.at(it->second))
94  % LoggingUtils::ChannelTypeToString(it->first));
95  if (channelManager_ != nullptr) {
96  channelManager_->onServerUpdated(getServerInfoByChannelType(collectionIt->second.at(it->second), it->first));
97  } else {
98  KAA_LOG_ERROR("Can not process server change. Channel manager was not specified");
99  }
100  } else {
101  KAA_LOG_INFO(boost::format("No available server for channel type %1%.") % LoggingUtils::ChannelTypeToString(type));
102  bootstrapTransport_->sync();
103  }
104  } else {
105  throw KaaException("There are no available servers at the time");

138 void BootstrapManager::onServerListUpdated(const OperationsServerList& list)
139 {
140  boost::recursive_mutex::scoped_lock lock(guard_);
141 
142  KAA_LOG_INFO(boost::format("Received new operations servers list: %1%")
143  % LoggingUtils::OperationServerListToString(list));
144 
145  operationServerList_ = list.operationsServerArray;
146  operationServersIterators_.clear();
147  operationServers_.clear();
148 
149  for (OperationsServer server : operationServerList_) {
150  auto supportedChannels = server.supportedChannelsArray;
151  for (SupportedChannel channel : supportedChannels) {
152  switch (channel.channelType) {
153  case ChannelType::HTTP:
154  case ChannelType::HTTP_LP:
155  break;
156  default:
157  throw KaaException("Unsupported channel type");
158  }
159  auto it = operationServers_.find(channel.channelType);
160  if (it != operationServers_.end()) {
161  it->second.push_back(server);
162  } else {
163  std::vector<OperationsServer> singleServerList;
164  singleServerList.reserve(10);
165  singleServerList.push_back(server);
166  operationServers_.insert(std::make_pair(channel.channelType, singleServerList));
167  }
168  }
169  }
170 
171  for (auto it = operationServers_.begin(); it != operationServers_.end(); ++it) {
172  it->second.shrink_to_fit();
173  std::random_shuffle(it->second.begin(), it->second.end());
174  std::sort(it->second.begin(), it->second.end()
175  , [&](const OperationsServer& a, const OperationsServer& b) -> bool { return a.priority > b.priority; });
176 
177  KAA_LOG_FTRACE(boost::format("Operations servers by type %1% shuffled: %2%")
178  % LoggingUtils::ChannelTypeToString(it->first)
179  % LoggingUtils::OperationServerArrayToString(it->second)
180  );
181 
182  operationServersIterators_[it->first] = 0;
183  }
184 
185  KAA_LOG_INFO(boost::format("Processed new operations servers list. Going to use server %1%") % (serverToApply.empty() ? "null" : serverToApply));
186 
187  if (!serverToApply.empty()) {
188  const OperationsServer* ops = getOPSByDnsName(serverToApply);
189  KAA_LOG_DEBUG(boost::format("Found server by name %1%. Object is %2%") % serverToApply % ops);
190  if (ops != nullptr) {
191  serverToApply.clear();
192  notifyChannelManangerAboutServer(*ops);
193  }
194  } else {
195  for (const auto& typeToIteratorPair : operationServersIterators_) {
196  KAA_LOG_DEBUG(boost::format("Updating server type = %1%")
197  % LoggingUtils::ChannelTypeToString(typeToIteratorPair.first)
198  );
199  if (channelManager_ != nullptr) {
200  channelManager_->onServerUpdated(getServerInfoByChannelType(operationServers_[typeToIteratorPair.first].at(typeToIteratorPair.second), typeToIteratorPair.first));
201  } else {
202  KAA_LOG_ERROR("Can not process server change. Channel manager was not specified");
203  }
204  }
205  }
206 }
