34 void UuidProcessStrategy::run(boost::shared_ptr<ICommonRecord> parent, const std::string &field, const avro::GenericDatum &datum)
35 {
36  avro::GenericFixed uuid_field = datum.value<avro::GenericFixed>();
37  uuid_t uuid;
38  std::copy(uuid_field.value().begin(), uuid_field.value().end(), uuid.begin());
39 
40  bool need_to_subscribe = true;
41 
42  // check for uuid change
43  if (parent->hasField(field)) {
44  ICommonRecord::fields_type uuid_field = parent->getField(field);
45  if (uuid_field.get()) {
46  std::vector<boost::uint8_t> prev_uuid = boost::any_cast<std::vector<boost::uint8_t> >(uuid_field->getValue());
47  if (!prev_uuid.empty()) {
48  uuid_t old_uuid;
49  std::copy(prev_uuid.begin(), prev_uuid.end(), old_uuid.begin());
50  if (!std::equal(old_uuid.begin(), old_uuid.end(), uuid.begin())) {
51  if (isSubscribedFn_(old_uuid)) {
52  unsubscribeFn_(old_uuid);
53  }
54  } else {
55  need_to_subscribe = false;
56  }
57  }
58  }
59  }
60 
61  parent->setField(field, CommonTypesFactory::createCommon<avro::AVRO_FIXED>(datum));
62  parent->setUuid(uuid);
63  if (need_to_subscribe) {
64  subscribeFn_(uuid, parent);
65  }
66 }

68 void RecordProcessStrategy::run(boost::shared_ptr<ICommonRecord> parent, const std::string &field, const avro::GenericDatum &datum)
69 {
70  const avro::GenericRecord &rec = datum.value<avro::GenericRecord>();
71  size_t field_count = rec.fieldCount();
72  uuid_t empty_uuid;
73  boost::shared_ptr<ICommonRecord> value;
74 
75  if (isRootRecord_) {
76  value = parent;
77  } else {
78  if (parent->hasField(field)) {
79  if(CommonValueTools::isRecord(parent->getField(field))) {
80  if(CommonValueTools::getRecord(parent->getField(field)).getSchema()->name().fullname()
81  .compare(rec.schema()->name().fullname()) == 0) {
82  value = boost::dynamic_pointer_cast<ICommonRecord, ICommonValue>(parent->getField(field));
83  } else {
84  value = CommonTypesFactory::createCommonRecord(empty_uuid, rec.schema());
85  }
86  } else {
87  value = CommonTypesFactory::createCommonRecord(empty_uuid, rec.schema());
88  }
89  } else {
90  value = CommonTypesFactory::createCommonRecord(empty_uuid, rec.schema());
91  }
92  }
93 
94  for (size_t i = 0; i < field_count; ++i) {
95  const std::string& field_name = rec.schema()->nameAt(i);
96  const avro::GenericDatum& innerDatum = rec.fieldAt(i);
97 
98  if (AvroGenericUtils::isUnchanged(innerDatum)) {
99  continue;
100  }
101 
102  boost::scoped_ptr<FieldProcessor> fp(new FieldProcessor(value, field_name));
103  AbstractStrategy *strategy = NULL;
104  switch (innerDatum.type()) {
105  case avro::AVRO_RECORD: {
106  strategy = new RecordProcessStrategy(isSubscribedFn_, subscribeFn_, unsubscribeFn_);
107  break;
108  }
109  case avro::AVRO_ARRAY: {
110  if (parent->hasField(field_name)) {
111  value->setField(field_name, parent->getField(field_name));
112  }
113 
114  strategy = new ArrayProcessStrategy(isSubscribedFn_, subscribeFn_, unsubscribeFn_);
115  break;
116  }
117  case avro::AVRO_NULL: {
118  strategy = new NullProcessStrategy();
119  break;
120  }
121  case avro::AVRO_FIXED: {
122  if (AvroGenericUtils::isUuid(innerDatum)) {
123  strategy = new UuidProcessStrategy(isSubscribedFn_, subscribeFn_, unsubscribeFn_);
124  } else {
125  strategy = new CommonProcessStrategy();
126  }
127  break;
128  }
129  case avro::AVRO_ENUM: {
130  if (AvroGenericUtils::isReset(innerDatum)) {
131  fp.reset(new FieldProcessor(parent, field_name));
132  strategy = new ArrayResetStrategy(isSubscribedFn_, unsubscribeFn_);
133  } else {
134  strategy = new CommonProcessStrategy();
135  }
136  break;
137  }
138  case avro::AVRO_MAP:
139  case avro::AVRO_UNKNOWN: {
140  throw KaaException(boost::format("Unsupported field type %1%") % datum.type());
141  }
142  default: {
143  strategy = new CommonProcessStrategy();
144  break;
145  }
146  }
147  fp->setStrategy(strategy);
148  fp->process(innerDatum);
149  }
150  if (!isRootRecord_) {
151  parent->setField(field, value);
152  } else {
153  auto map_ = value->getFields();
154  for (auto it = map_.begin(); it != map_.end(); ++it) {
155  parent->setField(it->first, it->second);
156  }
157  }
158 }

170 void ArrayResetStrategy::unregisterRecord(ICommonRecord &record)
171 {
172  auto map = record.getFields();
173  if (record.hasField("__uuid")) {
174  uuid_t uuid;
175  auto uuid_field = record.getField("__uuid");
176  std::vector<boost::uint8_t> uuid_raw = boost::any_cast<std::vector<boost::uint8_t> >(uuid_field->getValue());
177  std::copy(uuid_raw.begin(), uuid_raw.end(), uuid.begin());
178  if (isSubscribedFn_(uuid)) {
179  unsubscribeFn_(uuid);
180  }
181  }
182 
183  for (auto it = map.begin(); it != map.end(); ++it) {
184  switch (it->second->getCommonType()) {
185  case CommonValueType::COMMON_RECORD: {
186  unregisterRecord(dynamic_cast<ICommonRecord &>(*(*it).second));
187  break;
188  }
189  case CommonValueType::COMMON_ARRAY: {
190  unregisterArray(dynamic_cast<ICommonArray &>(*(*it).second));
191  break;
192  }
193  default:
194  break;
195  }
196  }

220 void ArrayProcessStrategy::run(boost::shared_ptr<ICommonRecord> parent, const std::string &field, const avro::GenericDatum &datum)
221 {
222  boost::shared_ptr<ICommonValue> commonValue;
223  boost::shared_ptr<ICommonArray> commonArray;
224  if (parent->hasField(field) && CommonValueTools::isArray(parent->getField(field))) {
225  commonValue = parent->getField(field);
226  commonArray.reset(new CommonArray(*dynamic_cast<CommonArray *>(commonValue.get())));
227  } else {
228  commonArray = CommonTypesFactory::createCommonArray(datum.value<avro::GenericArray>().schema());
229  }
230 
231  avro::GenericArray array = datum.value<avro::GenericArray>();
232  std::vector<avro::GenericDatum> vec = array.value();
233 
234  if (!vec.empty()) {
235  auto first_element = vec.begin();
236  if (AvroGenericUtils::isUuid(*first_element)) {
237  for (auto it = first_element; it != vec.end(); ++it) {
238  avro::GenericFixed uuid_field = (*it).value<avro::GenericFixed>();
239  uuid_t uuid;
240  std::copy(uuid_field.value().begin(), uuid_field.value().end(), uuid.begin());
241  ICommonArray::container_type & inner_list = commonArray->getList();
242  for (auto record_it = inner_list.begin(); record_it != inner_list.end(); ++record_it) {
243  ICommonArray::elements_type record = *record_it;
244  ICommonRecord * r_ptr = dynamic_cast<ICommonRecord *>(record.get());
245  if (r_ptr != NULL) {
246  if (uuid == r_ptr->getUuid()) {
247  ArrayResetStrategy ars(isSubscribedFn_, unsubscribeFn_);
248  ars.unregisterRecord(*r_ptr);
249  commonArray->getList().erase(record_it);
250  break;
251  }
252  }
253  }
254  }
255  } else {
256  for (auto it = first_element; it != vec.end(); ++it) {
257  const avro::GenericDatum & innerDatum = *it;
258  uuid_t empty_uuid;
259  boost::shared_ptr<ICommonRecord> record = CommonTypesFactory::createCommonRecord(empty_uuid, datum.value<avro::GenericArray>().schema());
260  boost::scoped_ptr<FieldProcessor> fp(new FieldProcessor(record, array_holder_field));
261  AbstractStrategy *strategy = NULL;
262  switch (innerDatum.type()) {
263  case avro::AVRO_RECORD: {
264  strategy = new RecordProcessStrategy(isSubscribedFn_, subscribeFn_, unsubscribeFn_);
265  break;
266  }
267  case avro::AVRO_ARRAY: {
268  strategy = new ArrayProcessStrategy(isSubscribedFn_, subscribeFn_, unsubscribeFn_);
269  break;
270  }
271  case avro::AVRO_NULL: {
272  strategy = new NullProcessStrategy();
273  break;
274  }
275  case avro::AVRO_MAP:
276  case avro::AVRO_UNKNOWN: {
277  throw KaaException(boost::format("Unsupported field type %1%") % datum.type());
278  }
279  default: {
280  strategy = new CommonProcessStrategy();
281  break;
282  }
283  }
284  fp->setStrategy(strategy);
285  fp->process(innerDatum);
286 
287  commonArray->getList().push_back(record->getField(array_holder_field));
288  }
289  }
290  parent->setField(field, commonArray);
291  }
