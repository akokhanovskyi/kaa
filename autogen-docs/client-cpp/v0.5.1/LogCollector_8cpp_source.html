58 void LogCollector::makeLogRecord(const LogRecord& record)
59 {
60  KAA_MUTEX_LOCKING("storageGuard_");
61  lock_type lock(storageGuard_);
62  KAA_MUTEX_LOCKED("storageGuard_");
63 
64  if (storage_ != nullptr) {
65  storage_->addLogRecord(record);
66  if (strategy_ != nullptr) {
67  LogUploadStrategyDecision decision = strategy_->isUploadNeeded(configuration_, status_);
68  switch (decision) {
69  case LogUploadStrategyDecision::CLEANUP: {
70  KAA_LOG_INFO(boost::format("Need to cleanup log storage."));
71  doCleanup();
72  break;
73  }
74  case LogUploadStrategyDecision::UPLOAD: {
75  if (!isUploading_) {
76  KAA_LOG_INFO(boost::format("Going to start logs upload."));
77  isUploading_ = true;
78  doUpload();
79  }
80  break;
81  }
82  case LogUploadStrategyDecision::NOOP:
83  default:
84  KAA_LOG_DEBUG(boost::format("Nothing to do now."));
85  break;
86  }
87  } else {
88  KAA_LOG_ERROR(boost::format("Can not decide if log upload is needed. Strategy: %1%, Configuration: %2%, Status: %3%)")
89  % strategy_ % configuration_ % status_);
90  }
91  } else {
92  KAA_LOG_ERROR("Can not add log to an empty storage");
93  }
94 }

136 void LogCollector::doUpload()
137 {
138  size_t blockSize = configuration_->getBlockSize();
139  while (status_->getConsumedVolume() > 0) {
140  LogSyncRequest request;
141  std::string requestId;
142  UuidGenerator::generateUuid(requestId);
143  request.requestId.set_string(requestId);
144  ILogStorage::container_type recordsBlock = storage_->getRecordBlock(blockSize, request.requestId.get_string());
145  if (recordsBlock.empty()) {
146  break;
147  }
148  for (auto it = recordsBlock.begin(); it != recordsBlock.end(); ++it) {
149  LogEntry entry;
150  entry.data.assign(it->getData().begin(), it->getData().end());
151  request.logEntries.get_array().push_back(entry);
152  }
153  lock_type lock(requestsGuard_);
154  KAA_LOG_INFO(boost::format("Generated log upload request: id= %1%") % request.requestId.get_string());
155  requests_.insert(std::make_pair(request.requestId.get_string(), request));
156  }
157 
158  if (transport_ != nullptr) {
159  transport_->sync();
160  } else {
161  KAA_LOG_WARN("Log upload postponed: transport was not set");
162  }
163 }

174 LogSyncRequest LogCollector::getLogUploadRequest()
175 {
176  lock_type lock(requestsGuard_);
177  LogSyncRequest request;
178  request.requestId.set_null();
179  KAA_LOG_INFO(boost::format("Trying to populate log upload request. Have %1% requests") % requests_.size());
180  if (!requests_.empty()) {
181  auto it = requests_.begin();
182  request = it->second;
183  KAA_LOG_INFO(boost::format("Added log upload request id %1%") % it->second.requestId.get_string());
184  requests_.erase(requests_.begin());
185  }
186  return request;
187 }

189 void LogCollector::onLogUploadResponse(const LogSyncResponse& response)
190 {
191  if (response.result == SyncResponseResultType::SUCCESS) {
192  storage_->removeRecordBlock(response.requestId);
193  if (!requests_.empty() || strategy_->isUploadNeeded(configuration_, status_) == LogUploadStrategyDecision::UPLOAD) {
194  doUpload();
195  } else {
196  isUploading_ = false;
197  }
198  } else {
199  KAA_LOG_ERROR("Failed to upload logs. Try again later.");
200  isUploading_ = false;
201  storage_->notifyUploadFailed(response.requestId);
202  }
203 }
