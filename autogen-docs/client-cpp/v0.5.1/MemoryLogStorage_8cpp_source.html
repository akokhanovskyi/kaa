22 void MemoryLogStorage::addLogRecord(const LogRecord & record)
23 {
24  LogBlock & block = logBlocks_.back();
25  if (block.finalized_ || block.actualSize_ + record.getSize() > blockSize_) {

27  LogBlock newBlock(blockSize_);
28  newBlock.logs_.push_back(record);

35  }
36  occupiedSize_ += record.getSize();

38  KAA_LOG_DEBUG(boost::format("Added log record (size: %1% bytes). Occupied size: %2% bytes")

93 void MemoryLogStorage::removeOldestRecords(size_t allowedVolume)
94 {
95  KAA_LOG_INFO(boost::format("Going to perform clean-up. Occupied %1% bytes, allowed %2% bytes") % occupiedSize_ % allowedVolume);
96  size_t releasedSpace = 0;
97  for (auto block = logBlocks_.begin(); block != logBlocks_.end(); ++block) {
98  if (!block->finalized_) {
99  while (occupiedSize_ > allowedVolume && !block->logs_.empty()) {
100  auto log_it = block->logs_.begin();
101  size_t frontLogSize = log_it->getSize();
102  block->actualSize_ -= frontLogSize;
103  occupiedSize_ -= frontLogSize;
104  releasedSpace += frontLogSize;
105  block->logs_.pop_front();
106  if (occupiedSize_ <= allowedVolume) {
107  KAA_LOG_INFO(boost::format("Released %1% bytes. Occupied %2% bytes, allowed %3% bytes")
108  % releasedSpace % occupiedSize_ % allowedVolume);
109  return;
110  }
111  }
112  }
113  }
114  KAA_LOG_ERROR(boost::format("Can not release enough space. "
115  "Released %1% bytes. Occupied %2% bytes"
116  ", allowed %3% bytes")
117  % releasedSpace % occupiedSize_ % allowedVolume);

138 void MemoryLogStorage::resize(size_t blockSize)
139 {
140  container_type logsPool;
141  for (auto it = logBlocks_.begin(); it != logBlocks_.end(); ++it) {
142  if (it->finalized_) {
143  continue;
144  }
145  logsPool.insert(logsPool.end(), it->logs_.begin(), it->logs_.end());
146  logBlocks_.erase(it);
147  }
148 
149  LogBlock tempBlock(blockSize);
150  for (auto it = logsPool.begin(); it != logsPool.end(); ++it) {
151  if (tempBlock.actualSize_ + it->getSize() >= blockSize) {
152  logBlocks_.push_back(tempBlock);
153  tempBlock = LogBlock(blockSize);
154  } else {
155  tempBlock.logs_.push_back(*it);
