47 std::vector<boost::uint8_t> OperationsDataProcessor::compileRequest(const std::map<TransportType, ChannelDirection>& transportTypes)
48 {
49  SyncRequest request;
50  request.requestId.set_int(++requestId);
51  request.configurationSyncRequest.set_null();
52  request.eventSyncRequest.set_null();
53  request.logSyncRequest.set_null();
54  request.notificationSyncRequest.set_null();
55  request.profileSyncRequest.set_null();
56  request.userSyncRequest.set_null();
57 
58  KAA_LOG_DEBUG(boost::format("Compiling sync request. RequestId: %1%") % requestId);
59 
60  auto metaRequest = metaDataTransport_->createSyncRequestMetaData();
61  request.syncRequestMetaData.set_SyncRequestMetaData(*metaRequest);
62  KAA_LOG_DEBUG(boost::format("Compiled SyncRequestMetaData: %1%")
63  % LoggingUtils::MetaDataSyncRequestToString(request.syncRequestMetaData));
64 
65  for (const auto& t : transportTypes) {
66  bool isDownDirection = t.second == ChannelDirection::DOWN;
67  switch (t.first) {
68  case TransportType::PROFILE :
69  if (isDownDirection) {
70  request.profileSyncRequest.set_ProfileSyncRequest(ProfileSyncRequest());
71  } else if (profileTransport_) {
72  auto ptr = profileTransport_->createProfileRequest();
73  if (ptr) {
74  request.profileSyncRequest.set_ProfileSyncRequest(*ptr);
75  } else {
76  request.profileSyncRequest.set_null();
77  }
78  } else {
79  KAA_LOG_WARN("Profile transport was not specified.");
80  }
81  KAA_LOG_DEBUG(boost::format("Compiled ProfileSyncRequest: %1%")
82  % LoggingUtils::ProfileSyncRequestToString(request.profileSyncRequest));
83  break;
84  case TransportType::CONFIGURATION:
85  if (isDownDirection) {
86  request.configurationSyncRequest.set_ConfigurationSyncRequest(ConfigurationSyncRequest());
87  } else if (configurationTransport_) {
88  auto ptr = configurationTransport_->createConfigurationRequest();
89  if (ptr) {
90  request.configurationSyncRequest.set_ConfigurationSyncRequest(*ptr);
91  } else {
92  request.configurationSyncRequest.set_null();
93  }
94  } else {
95  KAA_LOG_WARN("Configuration transport was not specified.");
96  }
97  KAA_LOG_DEBUG(boost::format("Compiled ConfigurationSyncRequest: %1%")
98  % LoggingUtils::ConfigurationSyncRequestToString(request.configurationSyncRequest));
99  break;
100  case TransportType::NOTIFICATION:
101  if (isDownDirection) {
102  request.notificationSyncRequest.set_NotificationSyncRequest(NotificationSyncRequest());
103  } else if (notificationTransport_) {
104  auto ptr = notificationTransport_->createNotificationRequest();
105  if (ptr) {
106  request.notificationSyncRequest.set_NotificationSyncRequest(*ptr);
107  } else {
108  request.notificationSyncRequest.set_null();
109  }
110  } else {
111  KAA_LOG_WARN("Notification transport was not specified.");
112  }
113  KAA_LOG_DEBUG(boost::format("Compiled NotificationSyncRequest: %1%")
114  % LoggingUtils::NotificationSyncRequestToString(request.notificationSyncRequest));
115  break;
116  case TransportType::USER:
117  if (isDownDirection) {
118  request.userSyncRequest.set_UserSyncRequest(UserSyncRequest());
119  } else if (userTransport_) {
120  auto ptr = userTransport_->createUserRequest();
121  if (ptr) {
122  request.userSyncRequest.set_UserSyncRequest(*ptr);
123  } else {
124  request.userSyncRequest.set_null();
125  }
126  } else {
127  KAA_LOG_WARN("User transport was not specified.");
128  }
129  KAA_LOG_DEBUG(boost::format("Compiled UserSyncRequest: %1%")
130  % LoggingUtils::UserSyncRequestToString(request.userSyncRequest));
131  break;
132  case TransportType::EVENT:
133  if (isDownDirection) {
134  request.eventSyncRequest.set_EventSyncRequest(EventSyncRequest());
135  } else if (eventTransport_) {
136  auto ptr = eventTransport_->createEventRequest(requestId);
137  if (ptr) {
138  request.eventSyncRequest.set_EventSyncRequest(*ptr);
139  } else {
140  request.eventSyncRequest.set_null();
141  }
142  } else {
143  KAA_LOG_WARN("Event transport was not specified.");
144  }
145  KAA_LOG_DEBUG(boost::format("Compiled EventSyncRequest: %1%")
146  % LoggingUtils::EventSyncRequestToString(request.eventSyncRequest));
147  break;
148  case TransportType::LOGGING:
149  if (isDownDirection) {
150  request.logSyncRequest.set_LogSyncRequest(LogSyncRequest());
151  } else if (loggingTransport_) {
152  auto ptr = loggingTransport_->createLogSyncRequest();
153  if (ptr) {
154  request.logSyncRequest.set_LogSyncRequest(*ptr);
155  } else {
156  request.logSyncRequest.set_null();
157  }
158  } else {
159  KAA_LOG_WARN("Log upload transport was not specified.");
160  }
161  KAA_LOG_DEBUG(boost::format("Compiled LogSyncRequest: %1%")
162  % LoggingUtils::LogSyncRequestToString(request.logSyncRequest));
163  break;
164  default: break;
165  }
166  }
167  SharedDataBuffer buffer = requestConverter_.toByteArray(request);
168  return std::vector<boost::uint8_t>(buffer.first.get(), buffer.first.get() + buffer.second);
169 }

171 void OperationsDataProcessor::processResponse(const std::vector<boost::uint8_t> &response)
172 {
173  SyncResponse syncResponse = responseConverter_.fromByteArray(response.data(), response.size());
174  boost::int32_t requestId = syncResponse.requestId.is_null() ? -1 : syncResponse.requestId.get_int();
175  KAA_LOG_INFO(boost::format("Got SyncResponse: requestId: %1%")
176  % requestId );
177 
178  if (!syncResponse.profileSyncResponse.is_null() ) {
179  KAA_LOG_DEBUG(boost::format("Got ProfileSyncResponse: %1%")
180  % LoggingUtils::ProfileSyncResponseToString(syncResponse.profileSyncResponse));
181  if (profileTransport_) {
182  profileTransport_->onProfileResponse(syncResponse.profileSyncResponse.get_ProfileSyncResponse());
183  } else {
184  KAA_LOG_ERROR("Got profile sync response, but profile transport was not set!");
185  }
186  }
187 
188  if (!syncResponse.configurationSyncResponse.is_null() ) {
189  KAA_LOG_DEBUG(boost::format("Got ConfigurationSyncResponse: %1%")
190  % LoggingUtils::ConfigurationSyncResponseToString(syncResponse.configurationSyncResponse));
191  if (configurationTransport_) {
192  configurationTransport_->onConfigurationResponse(syncResponse.configurationSyncResponse.get_ConfigurationSyncResponse());
193  } else {
194  KAA_LOG_ERROR("Got configuration sync response, but configuration transport was not set!");
195  }
196  }
197 
198  if (eventTransport_) {
199  eventTransport_->onSyncResponseId(requestId);
200  if (!syncResponse.eventSyncResponse.is_null() ) {
201  KAA_LOG_DEBUG(boost::format("Got EventSyncResponse: %1%")
202  % LoggingUtils::EventSyncResponseToString(syncResponse.eventSyncResponse));
203  eventTransport_->onEventResponse(syncResponse.eventSyncResponse.get_EventSyncResponse());
204  }
205  } else {
206  KAA_LOG_ERROR("Event transport was not set!");
207  }
208 
209  if (!syncResponse.notificationSyncResponse.is_null() ) {
210  KAA_LOG_DEBUG(boost::format("Got NotificationSyncResponse: %1%")
211  % LoggingUtils::NotificationSyncResponseToString(syncResponse.notificationSyncResponse));
212  if (notificationTransport_) {
213  notificationTransport_->onNotificationResponse(syncResponse.notificationSyncResponse.get_NotificationSyncResponse());
214  } else {
215  KAA_LOG_ERROR("Got notification sync response, but notification transport was not set!");
216  }
217  }
218 
219  if (!syncResponse.userSyncResponse.is_null() ) {
220  KAA_LOG_DEBUG(boost::format("Got UserSyncResponse: %1%")
221  % LoggingUtils::UserSyncResponseToString(syncResponse.userSyncResponse));
222  if (userTransport_) {
223  userTransport_->onUserResponse(syncResponse.userSyncResponse.get_UserSyncResponse());
224  } else {
225  KAA_LOG_ERROR("Got user sync response, but user transport was not set!");
226  }
227  }
228 
229  if (!syncResponse.logSyncResponse.is_null() ) {
230  KAA_LOG_DEBUG(boost::format("Got LogSyncResponse: %1%")
231  % LoggingUtils::LogSyncResponseToString(syncResponse.logSyncResponse));
232  if (loggingTransport_) {
233  loggingTransport_->onLogSyncResponse(syncResponse.logSyncResponse.get_LogSyncResponse());
234  } else {
235  KAA_LOG_ERROR("Got log upload sync response, but logging transport was not set!");
236  }
237  }
238 
239  if (!syncResponse.redirectSyncResponse.is_null()) {
240  KAA_LOG_DEBUG(boost::format("Got RedirectSyncResponse: %1%")
241  % LoggingUtils::RedirectSyncResponseToString(syncResponse.redirectSyncResponse));
242  if (redirectionTransport_) {
243  redirectionTransport_->onRedirectionResponse(syncResponse.redirectSyncResponse.get_RedirectSyncResponse());
244  } else {
245  KAA_LOG_ERROR("Got redirection sync response, but redirection transport was not set!");
246  }
247  }
248 
249  KAA_LOG_DEBUG("Processed SyncResponse");
250  clientStatus_->save();
251 }
