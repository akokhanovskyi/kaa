39 void NotificationManager::updateTopicSubscriptions(const TopicSubscribers& subscribers)
40 {
41  for (const auto& subscriber : subscribers) {
42  if (subscriber.first.empty() || !subscriber.second.lisnener_) {
43  throw KaaException("Bad subscription data");
44  }
45 
46  KAA_MUTEX_LOCKING("topicsGuard_");
47  boost::mutex::scoped_lock topicsLock(topicsGuard_);
48  KAA_MUTEX_LOCKED("topicsGuard_");
49 
50  auto topicIt = topics_.find(subscriber.first);
51 
52  if (topicIt == topics_.end()) {
53  throw KaaException("Topic with id " + subscriber.first + " is not available");
54  }
55 
56  KAA_MUTEX_LOCKING("notificationListenersGuard_");
57  boost::mutex::scoped_lock listenersLock(notificationListenersGuard_);
58  KAA_MUTEX_LOCKED("notificationListenersGuard_");
59 
60  if (subscriber.second.action_ == TopicSubscriberInfo::ADD) {
61  auto& listeners = notificationListeners_[subscriber.first];
62 
63  if (!listeners) {
64  listeners.reset(new NotificationListeners());
65  }
66 
67  if (listeners->empty() && topicIt->second.subscriptionType == SubscriptionType::VOLUNTARY) {
68  updateSubscriptionInfo(subscriber.first, SubscriptionCommandType::ADD);
69  }
70 
71  const auto connection = boost::bind(&INotificationListener::onNotificationRaw
72  , subscriber.second.lisnener_, _1, _2);
73  /* Disconnecting for case whether listener was already subscribed, no effect otherwise */
74  listeners->disconnect(connection);
75  listeners->connect(connection);
76  } else {
77  auto listenersIt = notificationListeners_.find(subscriber.first);
78

156 void NotificationManager::notificationReceived(const Notifications& notifications)
157 {
158  for (const Notification& notification : notifications) {
159  KAA_MUTEX_LOCKING("topicsGuard_");
160  boost::mutex::scoped_lock lock(topicsGuard_);
161  KAA_MUTEX_LOCKED("topicsGuard_");
162 
163  auto topicIt = topics_.find(notification.topicId);
164 
165  if (topicIt != topics_.end() && topicIt->second.id == notification.topicId) {
166  bool hasOwner = false;
167 
168  {
169  KAA_MUTEX_LOCKING("notificationListenersGuard_");
170  boost::mutex::scoped_lock notificationListenersLock(notificationListenersGuard_);
171  KAA_MUTEX_LOCKED("notificationListenersGuard_");
172 
173  auto listenerIt = notificationListeners_.find(notification.topicId);
174 
175  if (listenerIt != notificationListeners_.end() && !listenerIt->second->empty()) {
176  hasOwner = true;
177  (*listenerIt->second)(notification.topicId, notification.body);
178  }
179  }
180 
181  if (!hasOwner) {
182  mandatoryListeners_(notification.topicId, notification.body);
183  }
