28 {

32 
33  if (!schema_.get()) {

38 
39  AvroByteArrayConverter<avro::GenericDatum> converter;

42 
43  if (datumArray.type() != avro::AVRO_ARRAY) {

45  }
46 
47  const avro::GenericArray &deltaArray = datumArray.value<avro::GenericArray>();
48  const std::vector<avro::GenericDatum> &array = deltaArray.value();
49 
50  KAA_LOG_DEBUG(boost::format("Deltas count is %1%") % array.size());
51 
52  for (auto it = array.begin(); it != array.end(); ++it) {
53  const avro::GenericRecord &record = it->value<avro::GenericRecord>();
54  const avro::GenericDatum &datum = record.field("delta");

56  deltaReceivers_(index, datum, full_resync);
57  }
