41 void SchemaPersistanceManager::onSchemaUpdated(boost::shared_ptr<avro::ValidSchema> schema)
42 {
43  if (!schema.get()) {
44  throw KaaException("Empty schema was given");
45  }
46 
47  KAA_LOG_DEBUG(boost::format("Going to pass schema to storage %1%") % storage_);
48 
49  if (storage_) {
50  if (ignoreSchemaUpdate_) {
51  ignoreSchemaUpdate_ = false;
52  return;
53  }
54 
55  KAA_MUTEX_LOCKING("schemaPersistanceGuard_")
56  lock_type lock(schemaPersistanceGuard_);
57  KAA_MUTEX_LOCKED("schemaPersistanceGuard_")
58 
59  std::ostringstream osstr;
60  schema->toJson(osstr);

62 
63  ISchemaStorage::byte_buffer bytes;
64  for (auto it = str.begin(); it != str.end(); ++it) {
65  /*
66  * workaround for AVRO C++ issue
67  * https://issues.apache.org/jira/browse/AVRO-1352
68  */
69  if (*it == ',') {
70  auto it2 = it + 1;
71  while (it2 != str.end() && (isspace(*it2))) {
72  ++it2;
73  }
74  if (it2 != str.end() && *it2 == '}') {
75  continue;
76  }
77  } // end workaround
78 
79  bytes.push_back(*it);
80  }
81  storage_->saveSchema(bytes);
82  }
