70 void NotificationTransport::onNotificationResponse(const NotificationSyncResponse& response)
71 {
72  subscriptions_.clear();
73  acceptedUnicastNotificationIds_.clear();
74  clientStatus_->setNotificationSequenceNumber(response.appStateSeqNumber);
75 
76  DetailedTopicStates detailedStatesContainer = clientStatus_->getTopicStates();
77 
78  if (!response.availableTopics.is_null()) {
79  const auto& topics = response.availableTopics.get_array();
80 
81  detailedStatesContainer.clear();
82 
83  for (const auto& topic : topics) {
84  DetailedTopicState dts;
85 
86  dts.topicId = topic.id;
87  dts.topicName = topic.name;
88  dts.subscriptionType = topic.subscriptionType;
89  dts.sequenceNumber = 0;
90 
91  auto insertResult = notificationSubscriptions_.insert(std::make_pair(topic.id, 0));
92  if (!insertResult.second) {
93  dts.sequenceNumber = notificationSubscriptions_[topic.id];
94  }
95 
96  detailedStatesContainer[topic.id] = dts;
97  }
98 
99  if (notificationProcessor_ != nullptr) {
100  notificationProcessor_->topicsListUpdated(topics);
101  }
102  }
103 
104  if (!response.notifications.is_null()) {
105  const auto& notifications = response.notifications.get_array();
106 
107  for (const auto& notification : notifications) {
108  if (notification.uid.is_null()) {
109  auto& sequenceNumber = notificationSubscriptions_[notification.topicId];
110  boost::int32_t notificationSequenceNumber =
111  (notification.seqNumber.is_null()) ? 0 : notification.seqNumber.get_int();
112 
113  if (sequenceNumber > 0) {
114  sequenceNumber = std::max(notificationSequenceNumber, sequenceNumber);
115  } else {
116  sequenceNumber = notificationSequenceNumber;
117  }
118  detailedStatesContainer[notification.topicId].sequenceNumber = sequenceNumber;
119  } else {
120  const std::string& uid = notification.uid.get_string();
121  KAA_LOG_INFO(boost::format("Adding '%1%' to unicast accepted notifications") % uid);
122  acceptedUnicastNotificationIds_.push_back(uid);
123  }
124  }
125 
126  if (notificationProcessor_ != nullptr) {
127  notificationProcessor_->notificationReceived(notifications);
128  }
129  }
130 
131  clientStatus_->setTopicStates(detailedStatesContainer);
132 }
