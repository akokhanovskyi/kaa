69 static std::string convertToByteArrayString(const std::string & str)
70 {
71  const boost::uint8_t * bytes = reinterpret_cast<const boost::uint8_t *>(str.data());
72  std::size_t length = str.length();
73 
74  std::stringstream ss;
75  for (std::size_t i = 0; i < length; ++i) {
76  if (i != 0) {
77  ss << " ";
78  }
79  ss << std::setw(2) << std::setfill('0') << std::hex << (int)bytes[i];

239 template<>
240 void ClientParameter<DetailedTopicStates>::read(const std::string &strValue)
241 {
242  value_.clear();
243 
244  if (!strValue.empty()) {
245  std::size_t begin_pos = 0;
246  bool parse_more = true;
247  do {
248  std::size_t open_brace_pos = strValue.find_first_of('[', begin_pos);
249  std::size_t close_brace_pos = strValue.find_first_of(']', begin_pos);
250 
251  std::size_t comma1pos = strValue.find_first_of(',', open_brace_pos);
252  std::size_t comma2pos = strValue.find_first_of(',', comma1pos + 1);
253  std::size_t comma3pos = strValue.find_first_of(',', comma2pos + 1);
254 
255  if (comma1pos == std::string::npos || comma2pos == std::string::npos || comma3pos == std::string::npos) {
256  break;
257  }
258 
259  std::string topicId = strValue.substr(open_brace_pos+1, comma1pos - open_brace_pos - 1 );
260  std::string topicName = strValue.substr(comma1pos+1, comma2pos - comma1pos - 1) ;
261  std::string sType = strValue.substr(comma2pos+1, comma3pos - comma2pos - 1);
262  std::string topicSeqN = strValue.substr(comma3pos+1, close_brace_pos - comma3pos - 1);
263 
264  std::size_t commapos = strValue.find_first_of(",", close_brace_pos);
265  parse_more = commapos != std::string::npos;
266 
267  int32_t topicSN = std::stoi(topicSeqN, nullptr, 10);
268 
269  begin_pos = close_brace_pos + 1;
270 
271  DetailedTopicState ts;
272  ts.topicId = convertFromByteArrayString(topicId);
273  ts.topicName = convertFromByteArrayString(topicName);
274  ts.subscriptionType = (sType.compare("m") == 0 ? SubscriptionType::MANDATORY : SubscriptionType::VOLUNTARY);
275  ts.sequenceNumber = topicSN;
276 
277  value_.insert(std::make_pair(ts.topicId, ts));
278  } while (parse_more);
279  }
