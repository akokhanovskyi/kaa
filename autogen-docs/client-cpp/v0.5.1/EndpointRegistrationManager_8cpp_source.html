86 void EndpointRegistrationManager::onEndpointsAttach(const std::vector<EndpointAttachResponse>& endpoints)
87 {
88  boost::mutex::scoped_lock endpointLock(endpointsGuard_);
89 
90  bool hasChanged = false;
91 
92  std::string externalUserId;
93  if (userAttachRequest_) {
94  externalUserId = userAttachRequest_->userExternalId;
95  }
96 
97  for (const auto& endpoint : endpoints) {
98  auto requestIt = attachingEndpoints_.find(endpoint.requestId);
99  if (requestIt != attachingEndpoints_.end()) {
100  if (endpoint.result == SyncResponseResultType::SUCCESS) {
101  if (!endpoint.endpointKeyHash.is_null()) {
102  attachedEndpoints_.insert(std::make_pair(
103  requestIt->second.endpointData_, endpoint.endpointKeyHash.get_string()));
104  hasChanged = true;
105  }
106  if (requestIt->second.listener_) {
107  requestIt->second.listener_->onAttachSuccess(externalUserId, endpointAccessToken_);
108  }
109  } else {
110  KAA_LOG_ERROR(boost::format("Failed to attach endpoint. Attach endpoint request id: %1%")
111  % LoggingUtils::ByteArrayToString(endpoint.requestId));
112  if (requestIt->second.listener_) {
113  requestIt->second.listener_->onAttachFailure();
114  }
115  }
116  attachingEndpoints_.erase(requestIt);
117  } else {
118  KAA_LOG_ERROR(boost::format("Failed to find endpoint attach request by id: %1%")
119  % LoggingUtils::ByteArrayToString(endpoint.requestId));
120  }
121  }
122 
123  if (hasChanged) {
124  status_->setAttachedEndpoints(attachedEndpoints_);
125  attachedEPListListeners(attachedEndpoints_);
126  }
127 }

129 void EndpointRegistrationManager::onEndpointsDetach(const std::vector<EndpointDetachResponse>& endpoints)
130 {
131  boost::mutex::scoped_lock endpointLock(endpointsGuard_);
132  bool hasChanges = false;
133  for (EndpointDetachResponse endpoint : endpoints) {
134  auto requestIt = detachingEndpoints_.find(endpoint.requestId);
135  if (requestIt != detachingEndpoints_.end()) {
136  if (endpoint.result == SyncResponseResultType::SUCCESS) {
137  std::string epHash = requestIt->second.endpointData_;
138  for (auto attachedEpIt = attachedEndpoints_.begin(); attachedEpIt != attachedEndpoints_.end(); ++attachedEpIt) {
139  if (attachedEpIt->second.compare(epHash) == 0) {
140  attachedEndpoints_.erase(attachedEpIt);
141  hasChanges = true;
142  break;
143  }
144  }
145  if (requestIt->second.listener_) {
146  requestIt->second.listener_->onDetachSuccess(endpointAccessToken_);
147  }
148  } else {
149  KAA_LOG_ERROR(boost::format("Failed to detach endpoint. Detach endpoint request id: %1%")
150  % LoggingUtils::ByteArrayToString(endpoint.requestId));
151  if (requestIt->second.listener_) {
152  requestIt->second.listener_->onDetachFailure();
153  }
154  }
155  detachingEndpoints_.erase(endpoint.requestId);
156  } else {
157  KAA_LOG_ERROR(boost::format("Failed to find endpoint detach request by id: %1%")
158  % LoggingUtils::ByteArrayToString(endpoint.requestId));
159  }
160  }
161 
162  if (hasChanges) {
163  status_->setAttachedEndpoints(attachedEndpoints_);
164  attachedEPListListeners(attachedEndpoints_);
165  }
166 }

212 {
