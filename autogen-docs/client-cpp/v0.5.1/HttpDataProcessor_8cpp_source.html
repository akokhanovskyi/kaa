21 void HttpDataProcessor::verifyResponse(const IHttpResponse& response)
22 {
23  if (response.getStatusCode() != 200) {
24  throw TransportException(boost::format("Invalid response code %1%") % response.getStatusCode());
25  }
26  SharedBody rawResponse = response.getBody();
27  const std::string& signature = response.getHeaderField("X-SIGNATURE");
28  boost::scoped_array<boost::uint8_t> decodedSignature(new boost::uint8_t[signature.length() / 3 * 4]);
29  size_t sigLength = Botan::base64_decode(decodedSignature.get(), signature);
30  if (!encDec_->verifySignature(rawResponse.first.get(), rawResponse.second, decodedSignature.get(), sigLength)) {
31  throw TransportException("Failed to verify signature");
32  }
33 }
